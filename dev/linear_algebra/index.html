Linear Algebra

using Giac, LinearAlgebra

A = GiacMatrix([1 2; 3 4])
det(A)        # -2
tr(A)         # 5
inv(A)        # inverse matrix
transpose(A)  # transposed matrix

# Symbolic matrix
@giac_var a b c d
B = GiacMatrix([[a, b],
                [c, d]])
det(B)  # a*d-b*c

@giac_several_vars m 2 2
M = GiacMatrix([[m11, m12],
                [m21, m22]])
det(M)  # m11*m22-m12*m21
